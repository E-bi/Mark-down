```python
#![路径] python3
# 上面是用于声明解释器,方便终端直接启动,如:#!/user/allen/python python3 启东时: ./httpserver.py
"""
httpserver3.0
获取http请求
解析http请求
将请求发送给WebFrame
从WebFrame接受反馈数据
将数据组织为Response格式发送给客户端
"""
from socket import *
import sys
from threading import Thread
from config import *
import re
import json


# 和frame进行交互
def connect_frame(env):
    s = socket()
    try:
        # 连接webframe
        print('建立连接')
        # s.connect(('127.0.0.1', 8080))
        s.connect((frame_ip, frame_port))
        print('连接成功')
    except:
        return
    data = json.dumps(env)  # 转换为json
    print('发送请求')
    s.send(data.encode())  # 发送请求
    data = s.recv(1024 * 1024 * 10).decode()
    print('接受请求')
    try:
        data = json.loads(data)  # 约定格式{'status':200,'data':'c'}
        return data
    except:
        return


class HTTPServer:
    def __init__(self):
        self.host = host
        self.port = port
        self.address = (host, port)
        # 创建套接字
        self.create_socket()
        self.bind()

    # 创建套接字
    def create_socket(self):
        self.socketfd = socket()
        self.socketfd.setsockopt(SOL_SOCKET, SO_REUSEADDR, DEBUG)

    # 绑定地址
    def bind(self):
        self.socketfd.bind(self.address)

    # 启动服务
    def server_forever(self):
        self.socketfd.listen(5)
        print('Listen the port %d' % self.port)
        while True:
            connfd, addr = self.socketfd.accept()
            client = Thread(target=self.handle, args=(connfd,))
            client.setDaemon(True)
            client.start()

    # 具体处理客户端请求
    def handle(self, connfd):
        request = connfd.recv(4096).decode()
        # print(request)
        pattern = r'(?P<method>[A-Z]+)\s+(?P<info>/\S*)'
        try:
            env = re.match(pattern, request).groupdict()  # 形成捕获组字典
            print(env)
        except:
            connfd.close()
            return
        else:
            # 和frame进行交互
            response = connect_frame(env)
            if response:
                self.send_response(connfd, response)

    # 组织http响应,发送给浏览器
    def send_response(self, connfd, response):
        # response->{'status':'200','data':'ccc'}
        if response['status'] == '200':
            data = 'HTTP/1.1 200 OK\r\n'
            data += 'Content-Type:text/html\r\n'
            data += '\r\n'
            data += response['data']
        elif response['status'] == '404':
            data = 'HTTP/1.1 404 OK\r\n'
            data += 'Content-Type:text/html\r\n'
            data += '\r\n'
            data += response['data']
        elif response['status'] == '500':
            pass
        connfd.send(data.encode())


if __name__ == '__main__':
    httpd = HTTPServer()
    httpd.server_forever()  # 启动服务
```

```python
"""
wedframe
模拟网站的后端应用行为

从httpserver接收具体请求
根据请求进行逻辑处理和数据处理
将需要的数据反馈给httpserver
"""

from socket import *
import json
from settings import *
from multiprocessing import Process
from urls import *
from views import *

frame_address = (frame_ip, frame_port)


class Application:
    def __init__(self):
        self.sockfd = socket()
        self.sockfd.bind(frame_address)
        self.sockfd.setsockopt(SOL_SOCKET, SO_REUSEADDR, DEBUG)

    # 启动函数
    def start(self):
        self.sockfd.listen(5)
        print('Listen the port %d' % frame_port)
        while True:
            print('等待连接')
            connfd, addr = self.sockfd.accept()
            print('连接到:', addr)
            p = Process(target=self.handle, args=(connfd,))
            p.daemon = True
            p.start()

    # 处理具体的数据
    def handle(self, connfd):
        request = connfd.recv(1024).decode()
        # request->{"method": "GET", "info": "/"}
        request = json.loads(request)
        if request['method'] == 'GET':
            if request['info'] == '/' or request['info'][-5:] == '.html':
                response = self.get_html(request['info'])
            else:
                response = self.get_data(request['info'])
        elif request['method'] == 'POST':
            pass
        # 将结果返回给httpserver,有的网页数据无法json.dumps()
        try:
            response = json.dumps(response)
        except:
            connfd.close()
        else:
            connfd.send(response.encode())
            connfd.close()

    def get_html(self, info):
        if info == '/':
            filename = STATIC_DIR + '/bili.html'
        else:
            filename = STATIC_DIR + info
        try:
            fd = open(filename)
        except Exception:
            with open(STATIC_DIR + '/404.html') as f:
                return {'status': '404', 'data': f.read()}
        else:
            return {'status': '200', 'data': fd.read()}

    def get_data(self, info):
        for url, func in urls:
            if url == info:
                return {'status': '200', 'data': func()}
        return {'status': '404', 'data': 'Sorry'}


if __name__ == '__main__':
    app = Application()
    app.start()
```
